import { Application, Container, Graphics, Text, TextStyle } from 'pixi.js'; // 1. 게임 상태 관리 객체 const GameState = { coins: 100, gridSize: 4, spawnLevel: 1, board: [], // 2D Array get totalDamage() { return this.board.flat().reduce((sum, val) => sum + (val || 0), 0); }, get spawnProbability() { // 업그레이드 레벨별 소환 확률 표 const tables = { 1: { 2: 0.9, 4: 0.1 }, 2: { 2: 0.6, 4: 0.3, 8: 0.1 }, 3: { 2: 0.4, 4: 0.4, 8: 0.15, 16: 0.05 } }; return tables[this.spawnLevel] || tables[1]; } }; // 2. 메인 애플리케이션 초기화 (PixiJS v8 스타일) const app = new Application(); await app.init({ width: 480, height: 800, backgroundColor: 0x121214, antialias: true }); document.body.appendChild(app.canvas); // 3. 레이어 구성 (미니멀리즘 레이아웃) const shooterContainer = new Container(); const boardContainer = new Container(); app.stage.addChild(shooterContainer, boardContainer); // 위치 레이아웃 정렬 boardContainer.y = 450; // 4. 2048 보드 렌더링 로직 function initBoard() { GameState.board = Array(GameState.gridSize).fill(null) .map(() => Array(GameState.gridSize).fill(0)); drawBoard(); } function drawBoard() { boardContainer.removeChildren(); const cellSize = 70; const padding = 8; for (let r = 0; r < GameState.gridSize; r++) { for (let c = 0; c < GameState.gridSize; c++) { const value = GameState.board[r][c]; const cellBg = new Graphics(); // 미니멀 모노톤 스타일링 cellBg.roundRect( c * (cellSize + padding), r * (cellSize + padding), cellSize, cellSize, 8 ); cellBg.fill({ color: value ? 0x2A2A30 : 0x1A1A1E }); boardContainer.addChild(cellBg); if (value > 0) { const text = new Text({ text: value.toString(), style: { fontFamily: 'Arial', fontSize: 24, fill: 0xFFFFFF, fontWeight: 'bold' } }); text.anchor.set(0.5); text.x = c * (cellSize + padding) + cellSize / 2; text.y = r * (cellSize + padding) + cellSize / 2; boardContainer.addChild(text); } } } } // 5. 숫자가 소환될 때의 로직 function spawnTile() { const cost = 10; // 기본 소환 비용 if (GameState.coins < cost) return; // 빈 칸 찾기 const empties = []; for (let r = 0; r < GameState.gridSize; r++) { for (let c = 0; c < GameState.gridSize; c++) { if (GameState.board[r][c] === 0) empties.push({r, c}); } } if (empties.length === 0) return; // 보드가 가득 참 GameState.coins -= cost; const target = empties[Math.floor(Math.random() * empties.length)]; // 확률 표 기반 숫자 결정 GameState.board[target.r][target.c] = getWeightedRandomTile(); drawBoard(); updateUI(); } function getWeightedRandomTile() { const rand = Math.random(); let cumulative = 0; for (const [tile, prob] of Object.entries(GameState.spawnProbability)) { cumulative += prob; if (rand <= cumulative) return parseInt(tile); } return 2; } // 6. 슈팅 파트 - 플레이어 총 조작 (마우스/터치 연동) const gun = new Graphics(); gun.rect(-15, -10, 30, 20); gun.fill({ color: 0xFFFFFF }); gun.x = app.screen.width / 2; gun.y = 400; shooterContainer.addChild(gun); app.stage.eventMode = 'static'; app.stage.on('pointermove', (event) => { // 2048 보드 위 구역 안에서만 총이 좌우로 이동 gun.x = Math.max(30, Math.min(event.global.x, app.screen.width - 30)); }); // 7. 메인 게임 루프 (적 생성 및 충돌 검사) let enemies = []; let bullets = []; let spawnTimer = 0; app.ticker.add((ticker) => { // 자동 발사 로직 (보드 총합 데미지 반영) if (ticker.lastTime % 15 < 1) { fireBullet(); } // 적 이동 및 Bullet 충돌 정산 프로세스 updateProjectiles(ticker.deltaTime); }); function fireBullet() { const bullet = new Graphics(); bullet.circle(0, 0, 5); bullet.fill({ color: 0xFF0055 }); bullet.x = gun.x; bullet.y = gun.y - 10; bullet.damage = GameState.totalDamage; // 데미지 = 보드 총합 bullets.push(bullet); shooterContainer.addChild(bullet); } // 초기화 실행 initBoard();
